### Seed Generation (Cryptographic):
```javascript
// cyrb53 hash function
function cyrb53(str, seed = 0) {
    let h1 = 0xdeadbeef ^ seed, h2 = 0x41c6ce57 ^ seed;
    for(let i = 0, ch; i < str.length; i++) {
        ch = str.charCodeAt(i);
        h1 = Math.imul(h1 ^ ch, 2654435761);
        h2 = Math.imul(h2 ^ ch, 1597334677);
    }
    h1 = Math.imul(h1 ^ (h1 >>> 16), 2246822507);
    h1 ^= Math.imul(h2 ^ (h2 >>> 13), 3266489909);
    h2 = Math.imul(h2 ^ (h2 >>> 16), 2246822507);
    h2 ^= Math.imul(h1 ^ (h1 >>> 13), 3266489909);
    return 4294967296 * (2097151 & h2) + (h1 >>> 0);
}

// Seed = hash(name.toLowerCase() + ":" + attributes)
const seed = cyrb53("asad:Male, 28 years old, South Asian...");
// Result: 483839 (deterministic - same input = same output)
```

```javascript
// Build prompt with LOCKED character/environment attributes
const prompt = `
[CHARACTER]
${GLOBAL_STATE.characters.asad.attributes}

[ENVIRONMENT]
${GLOBAL_STATE.environments.drawing_room.attributes}

${sceneAction}
`;

// API Call with REFERENCE IMAGES (compulsory)
const formData = new FormData();
formData.append("prompt", prompt);
formData.append("model", settings.model);
formData.append("aspect_ratio", settings.aspectRatio);

// CRITICAL: Include master reference images
formData.append("ref_images", GLOBAL_STATE.characters.asad.master_image_url);
if (scene.characters.length > 1) {
    formData.append("ref_images", GLOBAL_STATE.characters.sara.master_image_url);
}
// MAX 2 ref_images per API call
```
